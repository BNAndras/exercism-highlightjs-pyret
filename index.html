include string-dict

data BinaryTree:
  | leaf
  | node(value :: A, left :: BinaryTree, right :: BinaryTree)
end

fun map-tree(f :: (A -> B), tree :: BinaryTree) -> BinaryTree:
  doc: "Map a function over a binary tree"
  cases (BinaryTree) tree:
    | leaf => leaf
    | node(v, l, r) =>
      node(f(v), map-tree(f, l), map-tree(f, r))
  end
where:
  t = node(1, leaf, node(2, leaf, leaf))
  map-tree(lam(x): x * 2 end, t) is node(2, leaf, node(4, leaf, leaf))
end
